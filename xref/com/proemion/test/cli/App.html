25  package com.proemion.test.cli;
26  
27  import com.jcabi.log.Logger;
28  import org.apache.commons.cli.CommandLine;
29  import org.apache.commons.cli.DefaultParser;
30  import org.apache.commons.cli.Options;
31  import org.apache.commons.cli.ParseException;
32  
33  /**
34   * Cli Entry Point.
35   * @author Armin Braun (armin.braun@proemion.com)
36   * @version $Id$
37   * @since 0.1
38   */
39  public final class App {
40  
41      /**
42       * Cassandra Host Cli Argument.
43       */
44      private static final String CASSANDRA = "cassandra_host";
45  
46      /**
47       * Migrations Directory Cli Argument.
48       */
49      private static final String MIGRATIONS = "migration_package";
50  
51      /**
52       * Cli Option Definitions.
53       */
54      private static final Options OPTIONS = new Options()
55          .addOption("c", App.CASSANDRA, true, "Cassandra Host")
56          .addOption("m", App.MIGRATIONS, true, "Migrations Package");
57  
58      /**
59       * Parsed CLI Arguments.
60       */
61      private final CommandLine cli;
62  
63      /**
64       * Ctor.
65       * @param args Cli arguments
66       * @throws ParseException On failure
67       */
68      private App(final String... args) throws ParseException {
69          this.cli = new DefaultParser().parse(App.OPTIONS, args);
70      }
71  
72      /**
73       * Entry Point.
74       * @param args Cli arguments
75       * @throws Exception On failure
76       */
77      public static void main(final String... args) throws Exception {
78          new App(args).exec();
79      }
80  
81      /**
82       * Run the application.
83       */
84      private void exec() {
85          if (this.cli.hasOption(App.CASSANDRA)) {
86              if (this.cli.hasOption(App.MIGRATIONS)) {
87                  Logger.info(App.class, "Not Implemented.");
88              } else {
89                  throw App.missing(App.MIGRATIONS);
90              }
91          } else {
92              throw App.missing(App.CASSANDRA);
93          }
94      }
95  
96      /**
97       * Builds IllegalArgumentException for missing cli argument.
98       * @param arg Argument missing
99       * @return Exception for missing argument
100      */
101     private static IllegalArgumentException missing(final String arg) {
102         return new IllegalArgumentException(
103             String.format("Argument %s missing!", arg)
104         );
105     }
106 }
